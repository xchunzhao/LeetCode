function climbStairs(n) {
  if (n === 1) return 1;
  if (n === 2) return 2;
  let prev = 1;
  let cur = 2;
  let temp;
  for(let i=3; i<=n; i++) {
    temp = prev + cur;
    prev = cur;
    cur = temp;
  }
  return temp;
}
